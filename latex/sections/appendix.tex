\section{Mã giả quy trình dự báo}

\begin{lstlisting}[style=pythonstyle, language=Python, caption={Mã giả quy trình dự báo điện năng}]
def predict_monthly_kwh(user_input):
    previous_kwh = user_input["kWh_previous"]
    num_people = user_input["num_people"]
    location = user_input["location"]
    devices = user_input["devices"]

    device_features = build_device_features(devices)
    weather_features = get_weather_features(location)
    time_features = build_time_features()

    features = merge_features(
        previous_kwh,
        num_people,
        device_features,
        weather_features,
        time_features
    )

    scaled_features = scaler.transform(features)
    predicted_scaled = anfis_model.predict(scaled_features)
    predicted_kwh = target_scaler.inverse_transform(predicted_scaled)

    explanation = explain_by_active_rules(anfis_model, scaled_features)
    return predicted_kwh, explanation
\end{lstlisting}

\section{Mẫu luật mờ}

\begin{itemize}[leftmargin=1.2cm]
  \item Nếu số điện tháng trước cao, nhiệt độ nóng và số thiết bị làm mát nhiều thì điện năng tháng tới cao.
  \item Nếu số điện tháng trước thấp, số người ít và tổng công suất thiết bị thấp thì điện năng tháng tới thấp.
  \item Nếu số điện tháng trước trung bình nhưng tháng dự báo nóng thì điện năng tháng tới có xu hướng tăng.
\end{itemize}
